fAdd xa xb {
    return xa + xb
}

fMultiply(xa, xb) {
    return xa * xb
}

fDivide xa xb {
    if (xa > xb) {
        xMax is xa
        xMin is xb
    } else {
        xMax is xb
        xMin is xa
    }
    return xMax / xMin
}

fMod(xa, xb) {
    if (xa > xb) {
        xMax is xa
        xMin is xb
    } else {
        xMax is xb
        xMin is xa
    }
    return xMax % xMin
}

fSubtract xa, xb {
    if (xa > xb) {
        xMax is xa
        xMin is xb
    } else {
        xMax is xb
        xMin is xa
    }
    return xMax - xMin
}
